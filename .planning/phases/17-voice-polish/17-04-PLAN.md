    ```python
    _SENTENCE_BOUNDARIES: tuple[str, ...] = (". ", "! ", "? ", "\n")

    def _split_at_boundary(buffer: str) -> tuple[str | None, str]:
        """Return (sentence, remainder). sentence is None if no boundary found."""
        # Find the EARLIEST boundary occurrence.
        best_idx = -1
        best_len = 0
        for marker in _SENTENCE_BOUNDARIES:
            idx = buffer.find(marker)
            if idx != -1 and (best_idx == -1 or idx < best_idx):
                best_idx = idx
                best_len = len(marker)
        if best_idx == -1:
            return None, buffer
        end = best_idx + best_len
        return buffer[:end], buffer[end:]
    ```

    ```python
        token_iter = astream_session(
            graph,
            user_input,
            config,

        # Producer/consumer bridge: tee the LLM token stream to (a) TTS and

                item = await token_queue.get()
                if item is None:

                _tts_iter(), turn_manager._stop_event
    ```
