   ```python
   from langgraph.graph import StateGraph, START, END
   from langgraph.checkpoint.memory import MemorySaver  # for testing
   from daily.orchestrator.state import SessionState

   def route_intent(state: SessionState) -> str:
       """Route based on last user message content."""
       last_msg = state.messages[-1].content.lower() if state.messages else ""
       summarise_keywords = ["summarise", "summarize", "summary", "thread", "email chain"]
       if any(kw in last_msg for kw in summarise_keywords):
           return "summarise_thread"
       return "respond"

   def build_graph(checkpointer=None):
       """Build and compile the orchestrator StateGraph.

       Args:
           checkpointer: LangGraph checkpointer instance. Use MemorySaver for tests,
                        AsyncPostgresSaver for production.
       Returns:
           CompiledStateGraph ready for ainvoke/astream.
       """
       from daily.orchestrator.nodes import respond_node, summarise_thread_node

       builder = StateGraph(SessionState)
       builder.add_node("respond", respond_node)
       builder.add_node("summarise_thread", summarise_thread_node)
       builder.add_conditional_edges(START, route_intent, {
           "respond": "respond",
           "summarise_thread": "summarise_thread",
       })
       builder.add_edge("respond", END)
       builder.add_edge("summarise_thread", END)
       return builder.compile(checkpointer=checkpointer)
   ```

   Per D-02: GPT-4.1 for summarise_thread (reasoning-heavy), GPT-4.1 mini for respond (quick follow-ups). Model selection happens inside the node functions, not in graph.py.

   ```python
       db_session: AsyncSession,
       session_date: date | None = None,
   ) -> dict:

       Per D-11: Reads from Redis cache, does NOT re-run pipeline.
       """

           "active_user_id": user_id,
           "preferences": preferences.model_dump(),
       }

       """Run a single turn through the orchestrator graph.

       Uses ainvoke (not invoke) to avoid hanging with async checkpointer (Pitfall 2).
       """
   ```

1. In `src/daily/cli.py`, add a `daily chat` command that runs an interactive orchestrator session with real adapters. Follow the same adapter instantiation pattern from `src/daily/briefing/scheduler.py` (resolve_pipeline_kwargs):

   ```python
   async def _resolve_email_adapters(user_id: int, settings) -> list:
       """Load integration tokens and instantiate real email adapters.

       Follows same pattern as briefing/scheduler.py resolve_pipeline_kwargs.
       """
       import base64
       from sqlalchemy import select
       from daily.db.engine import async_session
       from daily.db.models import IntegrationToken
       from daily.vault.crypto import decrypt_token
       from daily.integrations.google.adapter import GmailAdapter
       from daily.integrations.microsoft.adapter import OutlookAdapter

       vault_key = base64.b64decode(settings.vault_key) if settings.vault_key else b""
       adapters = []

       async with async_session() as session:
           result = await session.execute(
               select(IntegrationToken).where(IntegrationToken.user_id == user_id)
           )
           tokens = result.scalars().all()

       for token in tokens:
           decrypted = decrypt_token(token.encrypted_access_token, vault_key)
           if token.provider == "google":
               adapters.append(GmailAdapter(credentials=decrypted))
           elif token.provider == "microsoft":
               adapters.append(OutlookAdapter(credentials=decrypted))

       return adapters

   async def _run_chat_session(user_id: int = 1):
       """Async helper for interactive orchestrator chat session.

       Wires real email adapters from stored tokens so BRIEF-07 thread
       summarisation works end-to-end (per D-10).
       """
       from redis.asyncio import Redis
       from daily.config import Settings
       from daily.db.engine import async_session
       from daily.orchestrator.graph import build_graph
       from daily.orchestrator.session import (
           create_session_config,
           initialize_session_state,
           run_session,
           set_email_adapters,
       )

       settings = Settings()

       # 1. Instantiate real email adapters from stored tokens
       adapters = await _resolve_email_adapters(user_id, settings)
       set_email_adapters(adapters)

       # 2. Build graph (MemorySaver for Phase 3 CLI; AsyncPostgresSaver in Phase 5)
       from langgraph.checkpoint.memory import MemorySaver
       graph = build_graph(checkpointer=MemorySaver())

       # 3. Create session config and initial state
       config = await create_session_config(user_id)
       redis = Redis.from_url(settings.redis_url)
       async with async_session() as db_sess:
           initial_state = await initialize_session_state(user_id, redis, db_sess)
       await redis.aclose()

       # 4. Interactive loop
       print("dAIly chat session started. Type 'exit' or 'quit' to end.")
       if adapters:
           print(f"  {len(adapters)} email adapter(s) connected.")
       else:
           print("  No email adapters connected. Thread summaries won't work.")
           print("  Run 'daily connect gmail' or 'daily connect outlook' first.")
       print()

       first_turn = True
       while True:
           user_input = input("You: ").strip()
           if not user_input:
               continue
           if user_input.lower() in ("exit", "quit"):
               print("Session ended.")
               break

           result = await run_session(
               graph,
               user_input,
               config,
               initial_state=initial_state if first_turn else None,
           )
           first_turn = False

           # Extract last AI message from result
           messages = result.get("messages", [])
   ```

| Stored tokens -> adapter instantiation | Tokens decrypted in-memory only at adapter creation time, never logged |

|-----------|----------|-----------|-------------|-----------------|
| T-03-04 | Information Disclosure | thread_id | mitigate | thread_id = f"user-{user_id}-{date}" — user_id is system-assigned, never user-supplied. Prevents cross-user state access. |
