    ```python
    class TTSPipeline:
        def __init__(self, api_key: str, voice_id: str = DEFAULT_VOICE_ID):
            self._api_key = api_key
            self._voice_id = voice_id

        async def play_streaming(self, text: str, stop_event: asyncio.Event) -> None:
            """Stream TTS audio sentence-by-sentence with barge-in support.

            Per D-07: Split text into sentences, push each to Cartesia WebSocket,
            begin playback as first chunk arrives.
            Per D-04: Check stop_event between EVERY audio chunk, not just sentences.
            Per D-08: Use sounddevice RawOutputStream for PCM playback.

            Pitfall 5: Use try/finally to ensure sounddevice stream closes on cancel.
            """
    ```

    Implementation details:
    1. Create `AsyncCartesia(api_key=self._api_key)` client.
    2. Split text with `split_sentences(text)`.
    3. Open `client.tts.websocket_connect()` as async context manager.
    4. Create context with `connection.context(model_id="sonic-3", voice={"mode": "id", "id": self._voice_id}, output_format=CARTESIA_OUTPUT_FORMAT)`.

    6. Open `sd.RawOutputStream(samplerate=CARTESIA_SAMPLE_RATE, channels=1, dtype="float32")`.
    7. Iterate `async for response in ctx.receive()`: if `stop_event.is_set()` break immediately (D-04 per-chunk check). If `response.type == "chunk" and response.audio`, write to output stream.
    8. Wrap in try/finally to ensure output stream closes on CancelledError (Pitfall 5).
